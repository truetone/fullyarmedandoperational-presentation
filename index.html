Constructor

				
/* Create new instance */
var socket = new WebSocket("ws://localhost:9000");

/* Create new instance via secure connection. */
var secure_socket = new WebSocket("wss://localhost:9000");

Events: Open

Fires as soon as the server responds. 

				
socket.onopen = function(msg) {   
  $('#console').append('<p>Connection established.');
}

Events: Message

Fires upon receiving a message. 

				
socket.onmessage = function(msg) {
  $('#console').append('<p>Message received!');
}

Events: Error

Fires when there is an unexpected failure. 

				
socket.onerror = function(e) {
  $('#console').append('<p>Error!');
}

Events: Close

Fires when the connection closes. 

				
socket.onclose = function() {
  $('#console').append('<p>Connection closed.');
}

Methods: Send

Sends a message to the server. 

				
var msg = 'Hello World!';
socket.send(msg);

Methods: Close

Closes the connection. 

				
/* Optional arguments */
var close_code = 1000;
var close_msg = 'Closing normally. Nothing to see here.';

socket.close(close_code, close_msg);

Attributes: readyState

				
var state = socket.readyState;

switch (state) {
  case 0:
    console.log('Connecting');
  case 1:
    console.log('Open');
  case 2:
    console.log('Closing');
  case 3:
    console.log('Closed');
}

Attributes: bufferedAmount

Data in bytes queued to send to the server. 

				
var buffer_size = socket.bufferedAmount;

Attributes: Protocol

The sub-protocol chosen by the server. 

				
var protocol = socket.protocol;
